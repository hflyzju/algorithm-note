    def isMatch(self, s, p):
        """
        :type s: str
        :type p: str
        :rtype: bool
        """

        m, n = len(s), len(p)
        dp = [[False] * (n+1) for _ in range(m+1)]

        dp[0][0] = True

        for j in range(1, n+1):
            if p[j-1] == '*' and dp[0][j-1]:
                dp[0][j] = True
            else:
                break


        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if s[i-1] == p[j-1] or p[j-1] == '?':
                    if dp[i-1][j-1]:
                        dp[i][j] = True
                else:
                    if p[j-1] == '*':
                        if dp[i][j-1]:
                            dp[i][j] = True
                        if dp[i-1][j]:
                            dp[i][j] = True
        
        # for dpi in dp:

        return dp[m][n]
